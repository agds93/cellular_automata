##Pentomino-R:

#Su carta millimetrata:
gnuplot
unset key
unset title
set style line 11 lc rgb '#ffffff' lt 1
set border 0 back ls 11
set tics out nomirror scale 0,0.001
set format ''
set mxtics
set mytics
set style line 12 lc rgb '#ddccdd' lt 1 lw 1.5
set style line 13 lc rgb '#ddccdd' lt 1 lw 0.5
set grid xtics mxtics ytics mytics back ls 12, ls 13
set size square
plot "<(sed -n '1,5p' results.dat)" u 1:2 pt 7 ps 1 lc 'blue'
set term pdf
set out 'pentominoR.pdf'
replot
set out
q

#Cerchi nei quadrati (9x9):
gnuplot
unset key
unset title
set size square
set xrange [0:9]
set yrange [0:9]
set xtic format ""
set ytic format ""
set grid lt -1
set tics scale 0
plot "<(sed -n '1,5p' results.dat)" u ($1+0.5):($2+0.5) pt 7 ps 2 lc 'blue'
set term pdf
set out 'pentominoR.pdf'
replot
set out
q
